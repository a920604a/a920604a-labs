export function AuthProvider({ children }) {
    const [user, setUser] = useState(null);
    const [loading, setLoading] = useState(true);
    
    useEffect(() => {
        const unsubscribe = onAuthStateChanged(auth, (firebaseUser) => {
            if (firebaseUser) {
                setUser(firebaseUser);
            } else {
                setUser(null);
                navigate('/');
            }
            setLoading(false);
        });
        return unsubscribe;
    }, [navigate]);
}

export async function getHabits(userId) {
    const q = query(habitsCollection, where('userId', '==', userId));
    const snapshot = await getDocs(q);
    const habits = [];
    snapshot.forEach(doc => {
        habits.push({ id: doc.id, ...doc.data() });
    });
    return habits;
}

### 6. 成就系統實現

#### 成就計算邏輯
```typescript
// utils/achievementUtils.ts
export function checkConsecutiveDays(records: string[], targetDays: number): boolean {
    if (records.length < targetDays) return false;
    
    const sortedRecords = records.sort((a, b) => new Date(a).getTime() - new Date(b).getTime());
    let consecutive = 1;
    let maxConsecutive = 1;
    
    for (let i = 1; i < sortedRecords.length; i++) {
        const prevDate = new Date(sortedRecords[i - 1]);
        const currDate = new Date(sortedRecords[i]);
        const diffDays = (currDate.getTime() - prevDate.getTime()) / (1000 * 60 * 60 * 24);
        
        if (diffDays === 1) {
            consecutive++;
            maxConsecutive = Math.max(maxConsecutive, consecutive);
        } else {
            consecutive = 1;
        }
    }
    
    return maxConsecutive >= targetDays;
}
```

### 7. 資料匯出實現

#### CSV 匯出
```javascript
// utils/exportUtils.js
export function exportHabitsToCSV(habits, rangeText) {
    const headers = ['習慣名稱', '打卡次數', '最近打卡日期', '顏色'];
    const csvContent = [
        headers.join(','),
        ...habits.map(habit => [
            habit.name,
            habit.records.length,
            habit.records.length > 0 ? habit.records[habit.records.length - 1] : '無',
            habit.color
        ].join(','))
    ].join('\n');
    
    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = `習慣統計_${new Date().toISOString().split('T')[0]}.csv`;
    link.click();
}
```
